#### codeblockはこんな感じです

```js title="my-test-file.js"
console.log('hello world')
document.addEventListener("astro:page-load", () => {
  const btn_open = document.querySelector('[data-button="open"]');
  const btn_close = document.querySelector('[data-button="close"]');
  const nav = document.querySelector("[data-nav]");
  btn_open?.addEventListener("click", () => {
    nav?.classList.replace("-translate-x-full", "translate-x-0");
    if (nav?.classList.contains("invisible")) {
      nav?.classList.remove("invisible");
    }
  });
  btn_close?.addEventListener("click", () => {
    nav?.classList.replace("translate-x-0", "-translate-x-full");
  });
});
```
